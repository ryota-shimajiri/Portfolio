from datetime import date

class Human:
  def __init__(self):
    self.name = ''
    self.gender = 0
    self.job = ''
    self.language = ''
    
  def calculate_age(self, year, month, day):
    birth = date(year, month, day)
    today = date.today()
    age = today.year - birth.year
    # まだ今年の誕生日を迎えていないなら年齢を1減らす
    if (today.month, today.day) < (birth.month, birth.day):
      age -= 1
    return age
